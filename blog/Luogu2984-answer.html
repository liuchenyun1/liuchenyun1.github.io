题目描述

Farmer John有头奶牛，有个农场，编号，有条双向边,第条边连接农场和 ，该边的长度是 。
居住在农场的奶牛 ，它想送一份新年礼物给居住在农场 的奶牛，但是奶牛必须先到FJ(居住在编号的农场)那里取礼物，然后再送给奶牛。你的任务是：奶牛至少需要走多远的路程？

  2. 两遍，让牛去到FJ是一次，从FJ到牛再跑一次 于是就有了下面这份代码
#include<bits/stdc++.h>
using namespace std;
int N,M,B;
vector<pair<int,int> > G[50001];//first是长度，second是终点
queue<int> q;
bool inq[50001];
int d[50001];
void SPFA(int from){//SPFA
    memset(d,0x3f,sizeof(d));
    q.push(from);
    inq[from]=true;
    d[from]=0;
    while(!q.empty()){
        int t=q.front();
        q.pop();
        inq[t]=false;
        for(int i=0;i<G[t].size();i++){
            pair<int,int> t1=G[t][i]
            if(d[t]+t1.c<d[t1.second]){
                d[t1.second]=d[t]+t1.first;
                if(!inq[t1.second]){
                    q.push(t1.second);
                    inq[t1.second]=true;
                }
            }
        }
    }
    return;
}
int main(){
    scanf("%d%d%d",&N,&M,&B);
    for(int i=1;i<=M;i++){
        int u,v,w;
        scanf("%d%d%d",&u,&v,&w);
        G[u].push_back(make_pair(w,v));
        G[v].push_back(make_pair(w,u));//双向边
    }
    for(int i=1;i<=B;i++){
        int ca,cb;
        scanf("%d%d",&ca,&cb);
        SPFA(ca);//从农场到FJ
        int ans=d[1];
        SPFA(1);//从FJ到农场
        ans+=d[cb];
        printf("%d\n",ans);
    }
    return 0;
}

int B,N,M;

                    q.push(t1);
